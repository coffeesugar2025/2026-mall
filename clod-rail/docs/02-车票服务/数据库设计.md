# 车票服务数据库设计

## 车站表 (station)

```sql
CREATE TABLE station (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    station_code VARCHAR(10) UNIQUE NOT NULL COMMENT '车站代码',
    station_name VARCHAR(50) NOT NULL COMMENT '车站名称',
    short_name VARCHAR(20) COMMENT '简称',
    pinyin VARCHAR(100) COMMENT '拼音',
    city VARCHAR(50) COMMENT '所属城市',
    province VARCHAR(50) COMMENT '所属省份',
    station_type VARCHAR(20) COMMENT '车站类型',
    is_popular TINYINT DEFAULT 0 COMMENT '是否热门',
    status TINYINT DEFAULT 1 COMMENT '状态',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

## 线路表 (line)

```sql
CREATE TABLE line (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    line_code VARCHAR(20) UNIQUE NOT NULL COMMENT '线路编码',
    line_name VARCHAR(100) NOT NULL COMMENT '线路名称',
    start_station_id BIGINT COMMENT '起始站',
    end_station_id BIGINT COMMENT '终点站',
    total_distance INT COMMENT '总里程(km)',
    status TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

## 线路途经站表 (line_stations)

```sql
CREATE TABLE line_stations (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    line_id BIGINT NOT NULL,
    station_id BIGINT NOT NULL,
    sequence INT NOT NULL COMMENT '顺序',
    distance INT COMMENT '距起点距离(km)',
    FOREIGN KEY (line_id) REFERENCES line(id),
    FOREIGN KEY (station_id) REFERENCES station(id)
);
```

## 车次表 (train)

```sql
CREATE TABLE train (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    train_number VARCHAR(10) UNIQUE NOT NULL COMMENT '车次号',
    train_type VARCHAR(10) NOT NULL COMMENT '车型(G/D/K/T)',
    line_id BIGINT COMMENT '所属线路',
    total_seats INT NOT NULL COMMENT '总座位数',
    status TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (line_id) REFERENCES line(id)
);
```

## 车票表 (ticket)

```sql
CREATE TABLE ticket (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    train_id BIGINT NOT NULL,
    start_station_id BIGINT NOT NULL COMMENT '出发站',
    end_station_id BIGINT NOT NULL COMMENT '到达站',
    departure_time TIME NOT NULL COMMENT '出发时间',
    arrival_time TIME NOT NULL COMMENT '到达时间',
    departure_date DATE NOT NULL COMMENT '出发日期',
    price DECIMAL(10,2) NOT NULL COMMENT '票价',
    seat_type VARCHAR(20) NOT NULL COMMENT '座位类型',
    available_seats INT NOT NULL COMMENT '余票',
    is_hot TINYINT DEFAULT 0 COMMENT '是否热门',
    status TINYINT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (train_id) REFERENCES train(id)
);

CREATE INDEX idx_ticket_route ON ticket(start_station_id, end_station_id, departure_date);
```

## 座位表 (seat)

```sql
CREATE TABLE seat (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    ticket_id BIGINT NOT NULL,
    carriage_no VARCHAR(10) NOT NULL COMMENT '车厢号',
    seat_no VARCHAR(10) NOT NULL COMMENT '座位号',
    seat_type VARCHAR(20) NOT NULL COMMENT '座位类型',
    full_seat_code VARCHAR(20) GENERATED ALWAYS AS (CONCAT(carriage_no, '-', seat_no)) STORED,
    status TINYINT DEFAULT 0 COMMENT '0-可用 1-已占',
    FOREIGN KEY (ticket_id) REFERENCES ticket(id)
);

CREATE INDEX idx_seat_ticket ON seat(ticket_id, status);
```

## 停靠站表 (stop_stations)

```sql
CREATE TABLE stop_stations (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    train_id BIGINT NOT NULL,
    station_id BIGINT NOT NULL,
    sequence INT NOT NULL COMMENT '停靠顺序',
    arrive_time TIME COMMENT '到达时间',
    depart_time TIME COMMENT '出发时间',
    stay_minutes INT DEFAULT 0 COMMENT '停留时间(分钟)',
    FOREIGN KEY (train_id) REFERENCES train(id),
    FOREIGN KEY (station_id) REFERENCES station(id)
);
```
