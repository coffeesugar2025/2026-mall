# 积分商城数据库设计

## 积分商品表 (points_product)

```sql
CREATE TABLE points_product (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL COMMENT '商品名称',
    description TEXT COMMENT '商品描述',
    image_url VARCHAR(255) COMMENT '商品图片',
    category VARCHAR(50) COMMENT '商品分类',
    points_required INT NOT NULL COMMENT '所需积分',
    original_price DECIMAL(10,2) COMMENT '原价',
    stock INT DEFAULT 0 COMMENT '库存数量',
    exchange_count INT DEFAULT 0 COMMENT '兑换次数',
    valid_days INT DEFAULT 30 COMMENT '有效天数',
    exchange_limit_day INT DEFAULT 2 COMMENT '每日限兑',
    exchange_limit_month INT DEFAULT 10 COMMENT '每月限兑',
    status TINYINT DEFAULT 1 COMMENT '状态：1-上架，0-下架',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE INDEX idx_product_category ON points_product(category);
CREATE INDEX idx_product_status ON points_product(status);
```

## 积分兑换订单表 (mall_order)

```sql
CREATE TABLE mall_order (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_no VARCHAR(32) UNIQUE NOT NULL COMMENT '订单号',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    product_id BIGINT NOT NULL COMMENT '商品ID',
    product_name VARCHAR(100) NOT NULL COMMENT '商品名称',
    quantity INT DEFAULT 1 COMMENT '数量',
    points_used INT NOT NULL COMMENT '使用积分',
    status VARCHAR(20) DEFAULT 'PENDING' COMMENT '订单状态',
    receiver_name VARCHAR(50) COMMENT '收货人',
    receiver_phone VARCHAR(11) COMMENT '收货电话',
    receiver_address VARCHAR(200) COMMENT '收货地址',
    tracking_no VARCHAR(50) COMMENT '物流单号',
    shipped_at DATETIME COMMENT '发货时间',
    completed_at DATETIME COMMENT '完成时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (product_id) REFERENCES points_product(id)
);

CREATE INDEX idx_mall_order_user ON mall_order(user_id);
CREATE INDEX idx_mall_order_status ON mall_order(status);
```

## 积分记录表 (points_record)

```sql
CREATE TABLE points_record (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL COMMENT '用户ID',
    points_change INT NOT NULL COMMENT '积分变动',
    change_type VARCHAR(20) NOT NULL COMMENT '变动类型：earn/consume/expire',
    source VARCHAR(50) COMMENT '来源',
    description VARCHAR(200) COMMENT '描述',
    related_order_no VARCHAR(32) COMMENT '关联订单号',
    expire_time DATETIME COMMENT '过期时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_points_record_user ON points_record(user_id);
CREATE INDEX idx_points_record_type ON points_record(change_type);
```
